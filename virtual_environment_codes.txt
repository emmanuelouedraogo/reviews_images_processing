cd path/to/your/project  

*****************  Using venv   *****************

python -m venv myenv  

# Windows
myenv\Scripts\activate  

# Mac & linux
source myenv/bin/activate  


deactivate  

*****************  Using virtualenv *****************

pip install virtualenv  

virtualenv myenv 

myenv\Scripts\activate  

source myenv/bin/activate  

deactivate
